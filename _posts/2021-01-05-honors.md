<h3><strong>Outstanding Student Leader</strong></h3>
<!-- /wp:heading -->

<!-- wp:heading {"level":4} -->
<h4>Honor Date: Oct. 2016<br>Honor Issuer: Shandong University of Science and Technology</h4>
<!-- /wp:heading -->

<!-- wp:heading {"level":4} -->
<h4>Honor Description:</h4>
<!-- /wp:heading -->
